# P1197 [JSOI2008] 星球大战

$time:2025/6/8$

题目链接：[P1197 [JSOI2008] 星球大战](https://www.luogu.com.cn/problem/P1197)

## Solution

挺简单，主要是要想到要反着搞

从最后一个往前，一个一个连接就好做了

## Code

```c++
#include<bits/stdc++.h>
#define ll __int128
using namespace std;
const ll N = 1e6 + 100, M = 998244353, INF = 1e9 + 10;
int n, m, k,num, ne[N], to[N], fa[N], fi[N], tmx[N], a[N];
int p[N], cnt;
ll read() {
	ll x = 0, f = 1; char ch = getchar();
	while(ch < '0' || ch > '9') {if(ch == '-') f = -1; ch = getchar();}
	while(ch >= '0' && ch <= '9') {x = x * 10 + ch - '0'; ch = getchar();}
	return x * f;
}
void add(int u, int v) {
	ne[++num] = fi[u];
	fi[u] = num;
	to[num] = v;
}
int GetF(int x) {return x == fa[x] ? x:fa[x] = GetF(fa[x]); }
int main() {
	n = read(), m = read();
	for(int i = 1; i <= m; i++) {
		int u = read(), v = read();
		add(u, v), add(v, u);
	}
	k = read();
	for(int i = 0; i < n; i++) tmx[i] = k + 1;
	for(int i = k; i >= 1; i--) {
		int t = read();
		a[i] = t;
		tmx[t] = k - i + 1; 
	}
	for(int i = 0; i < n; i++) fa[i] = i;
	for(int i = 0; i < n; i++) {
		if(tmx[i] != k + 1) continue;
		for(int j = fi[i]; j; j = ne[j]) {
			int v = to[j];
			if(tmx[v] != k + 1) continue;
			int fx = GetF(i), fy = GetF(v);
			if(fx != fy) fa[fx] = fy;
		}
	}
	int ans = 0, ansx = 0;
	for(int i = 0; i < n; i++) {
		int fx = GetF(i);
		if(tmx[i] != k + 1) continue;
		if(fx == i) ans++;
	}
	ansx = ans;
	for(int j = 1; j <= k; j++) {
		int u = a[j];
		int num = 0;
		for(int i = fi[u]; i; i = ne[i]) {
			int v = to[i];
			if(tmx[v] < k - j + 1) continue;
			int fx = GetF(u), fy = GetF(v);
			if(fx != fy) {
				fa[fy] = fx;
				num++;
			}
		}
		ans -= (num - 1);
		p[++cnt] = ans;
	}
	for(int i = cnt; i >= 1; i--) cout << p[i] << endl;
	cout << ansx;
	return 0;
}
```
